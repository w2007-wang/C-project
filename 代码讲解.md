```cpp
struct Cell {
```

**初始化逻辑**：

```cpp
void GameBoard::initBoard() {
    // 1. 清空棋盘
    for (int y = 0; y < SIZE; ++y)
        for (int x = 0; x < SIZE; ++x)
            board[y][x] = Cell();
    
    // 2. 随机放置20个箭头（避开玩家起始位置）
    while (cnt < ARROW_COUNT) {
        int x = QRandomGenerator::global()->bounded(SIZE);
        int y = QRandomGenerator::global()->bounded(SIZE);
        if (!board[y][x].hasArrow && !(x==1&&y==1) && !(x==7&&y==7)) {
            board[y][x].hasArrow = true;
            board[y][x].dir = (Direction)QRandomGenerator::global()->bounded(DirCount);
            cnt++;
        }
    }
    
    // 3. 随机放置8个障碍物
    for (int i = 0; i < 8; ++i) {
        int x = QRandomGenerator::global()->bounded(SIZE);
        int y = QRandomGenerator::global()->bounded(SIZE);
        if (!board[y][x].hasArrow)
            board[y][x].isBlock = true;
    }
}
```

**交互方法**：

```cpp
// 点击旋转箭头（顺时针90度）
void GameBoard::rotateArrow(int x, int y) {
    if (board[y][x].hasArrow)
        board[y][x].dir = (Direction)((board[y][x].dir + 1) % DirCount);
}
```

```cpp
QPoint MainWindow::posToGrid(const QPoint &pos) const {
    QRect br = getBoardRect();
    int cs = getCellSize();
    int gx = (pos.x() - br.x()) / cs;
    int gy = (pos.y() - br.y()) / cs;
    // 边界限制（0-8）
    return QPoint(qBound(0, gx, 8), qBound(0, gy, 8));
}
```

```cpp
void MainWindow::updateMoveAnimation() {
    int oldX = currDoll->x;
    int oldY = currDoll->y;
    currDoll->moveStep();  // 移动一步
    
    int x = currDoll->x;
    int y = currDoll->y;
    Doll* enemy = (currPlayer == 1) ? doll2 : doll1;
    
    // 碰撞检测
    if (x < 0 || x >= 9 || y < 0 || y >= 9) {
        // 出界：回退位置，停止移动，切换玩家
        currDoll->x = oldX;
        currDoll->y = oldY;
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }
    
    Cell cell = board->getCell(x, y);
    
    if (cell.isBlock) {
        // 碰到障碍物：停止移动
        currDoll->x = oldX;
        currDoll->y = oldY;
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }
    
    if (x == enemy->x && y == enemy->y) {
        // 命中敌人：造成伤害
        int damage = currDoll->arrowCount * 2;  // 途经箭头数 × 2
        enemy->hurt(damage);
        
        if (doll1->isDead()) gameOver(2);  // 玩家1死亡，玩家2获胜
        if (doll2->isDead()) gameOver(1);  // 玩家2死亡，玩家1获胜
        
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }
    
    if (cell.hasArrow) {
        // 碰到箭头：改变方向，计数+1
        currDoll->currentDir = cell.dir;
        currDoll->arrowCount++;
    }
}
```

## 三、游戏规则总结

### 基本规则

1. **棋盘**：9×9 网格，随机分布箭头和障碍物
2. **玩家**：双人对战，红方（玩家1）从左上角开始，蓝方（玩家2）从右下角开始
3. **操作**：
   - **WASD**：选择发射方向
   - **空格键**：发射人偶
   - **鼠标点击**：旋转箭头方向

### 移动机制

1. 人偶沿发射方向直线移动
2. 碰到箭头时改变移动方向，并累计箭头数
3. 碰到障碍物或出界时停止移动
4. 命中敌人时造成伤害：**伤害 = 途经箭头数 × 2**

### 回合系统

- 每回合60秒时间限制
- 时间耗尽或移动结束后切换玩家
- 一方血量归零则游戏结束，另一方获胜

## 八、gameboard.cpp - 棋盘类实现

**功能**：实现棋盘的初始化和交互逻辑

```cpp
#include "gameboard.h"
#include <QRandomGenerator>

GameBoard::GameBoard(QObject *parent) : QObject(parent) {
    initBoard();  // 构造时初始化棋盘
}

void GameBoard::initBoard() {
    // 1. 清空所有单元格
    for (int y = 0; y < SIZE; ++y)
        for (int x = 0; x < SIZE; ++x)
            board[y][x] = Cell();

    // 2. 随机放置27个箭头（避开玩家起始位置）
    int cnt = 0;
    while (cnt < ARROW_COUNT) {
        int x = QRandomGenerator::global()->bounded(SIZE);
        int y = QRandomGenerator::global()->bounded(SIZE);
        // 条件：无箭头 && 不是玩家1起始点(1,1) && 不是玩家2起始点(7,7)
        if (!board[y][x].hasArrow && !(x==1&&y==1) && !(x==7&&y==7)) {
            board[y][x].hasArrow = true;
            board[y][x].dir = (Direction)QRandomGenerator::global()->bounded(DirCount);
            cnt++;
        }
    }

    // 3. 随机放置8个障碍物（放在无箭头的位置）
    for (int i = 0; i < 8; ++i) {
        int x = QRandomGenerator::global()->bounded(SIZE);
        int y = QRandomGenerator::global()->bounded(SIZE);
        if (!board[y][x].hasArrow)
            board[y][x].isBlock = true;
    }
}

void GameBoard::rotateArrow(int x, int y) {
    // 边界检查
    if (x < 0 || x >= SIZE || y < 0 || y >= SIZE) return;
    // 顺时针旋转90度
    if (board[y][x].hasArrow)
        board[y][x].dir = (Direction)((board[y][x].dir + 1) % DirCount);
}

Cell GameBoard::getCell(int x, int y) {
    // 边界检查，越界返回空单元格
    if (x < 0 || x >= SIZE || y < 0 || y >= SIZE) return Cell();
    return board[y][x];
}
```

```cpp
// 获取棋盘所在的矩形区域
QRect MainWindow::getBoardRect() const {
    int w = width();
    int h = height() - m_boardMargin;
    int boardPixel = qMin(w, h);  // 取宽高较小值，保持正方形
    int x = (w - boardPixel) / 2;
    int y = m_boardMargin / 2;
    return QRect(x, y, boardPixel, boardPixel);
}

// 获取单元格大小
int MainWindow::getCellSize() const {
    return getBoardRect().width() / 9;  // 9列均分
}

// 网格坐标 → 屏幕坐标（返回单元格中心）
QPoint MainWindow::gridToPos(int x, int y) const {
    QRect br = getBoardRect();
    int cs = getCellSize();
    return QPoint(br.x() + x * cs + cs/2, br.y() + y * cs + cs/2);
}

// 屏幕坐标 → 网格坐标
QPoint MainWindow::posToGrid(const QPoint &pos) const {
    QRect br = getBoardRect();
    int cs = getCellSize();
    if (cs <= 0) return QPoint(0,0);

    int gx = (pos.x() - br.x()) / cs;
    int gy = (pos.y() - br.y()) / cs;

    gx = qBound(0, gx, 8);  // 限制在0-8范围内
    gy = qBound(0, gy, 8);
    return QPoint(gx, gy);
}
```

```cpp
// 绘制棋盘
void MainWindow::drawBoard(QPainter &p) {
    QRect br = getBoardRect();
    int cs = getCellSize();

    for (int y = 0; y < 9; ++y) {
        for (int x = 0; x < 9; ++x) {
            QRect r(br.x() + x*cs, br.y() + y*cs, cs, cs);
            // 棋盘格子颜色交替
            p.setPen(Qt::black);
            p.setBrush((x + y) % 2 ? Qt::white : QColor(240,240,240));
            p.drawRect(r);

            Cell c = board->getCell(x, y);
            // 绘制障碍物（深灰色）
            if (c.isBlock) {
                p.setBrush(Qt::darkGray);
                p.drawRect(r.adjusted(4,4,-4,-4));
            }
            // 绘制箭头（蓝色线）
            if (c.hasArrow) {
                p.setPen(Qt::blue);
                QPoint ct = r.center();
                int len = cs * 0.25;
                switch (c.dir) {
                case Right: p.drawLine(ct, ct + QPoint(len,0)); break;
                case Down:  p.drawLine(ct, ct + QPoint(0,len)); break;
                case Left:  p.drawLine(ct, ct - QPoint(len,0)); break;
                case Up:    p.drawLine(ct, ct - QPoint(0,len)); break;
                }
            }
        }
    }
}
```

### 10.5 移动与碰撞逻辑

```cpp
void MainWindow::launch() {
    currDoll->arrowCount = 0;
    currDoll->applyLaunchDirection();  // 应用发射方向
    currDoll->isMoving = true;
    moveTimer->start();  // 启动移动动画
}

void MainWindow::updateMoveAnimation() {
    int oldX = currDoll->x;
    int oldY = currDoll->y;
    currDoll->moveStep();  // 移动一步
    int x = currDoll->x;
    int y = currDoll->y;

    Doll* enemy = (currPlayer == 1) ? doll2 : doll1;

    // 边界检测：出界则回退
    if (x < 0 || x >= 9 || y < 0 || y >= 9) {
        currDoll->x = oldX;
        currDoll->y = oldY;
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }

    Cell cell = board->getCell(x, y);

    // 障碍物检测
    if (cell.isBlock) {
        currDoll->x = oldX;
        currDoll->y = oldY;
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }

    // 敌人碰撞检测
    if (x == enemy->x && y == enemy->y) {
        currDoll->x = oldX;
        currDoll->y = oldY;

        int damage = currDoll->arrowCount * 2;  // 伤害 = 箭头数 × 2
        enemy->hurt(damage);

        QMessageBox::information(this, "命中",
            QString("途经箭头:%1 → 造成伤害:%2").arg(currDoll->arrowCount).arg(damage));

        // 判断游戏结束
        if (doll1->isDead()) gameOver(2);
        if (doll2->isDead()) gameOver(1);

        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }

    // 箭头检测：改变方向
    if (cell.hasArrow) {
        currDoll->currentDir = cell.dir;
        currDoll->arrowCount++;
    }

    update();  // 触发重绘
}
```

## 二、main.cpp —— 程序入口，一切从这里开始

```cpp
#include "mainwindow.h"
#include "StartWindow.h"
#include <QApplication>

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);

    StartWindow s;
    MainWindow w;

    // 点击开始 → 显示游戏窗口
    QObject::connect(&s, &StartWindow::startGameClicked, [&](){
        w.show();
    });

    s.show();     // 先显示开始界面
    return a.exec();
}
```

`main` 函数的内容非常精炼。

首先创建 `QApplication a(argc, argv)`，这是每个 Qt Widgets 应用的必有第一行。它初始化了 Qt 框架的底层资源——字体系统、配色方案、事件循环等。

接着实例化了两个窗口：`StartWindow s`（开始界面）和 `MainWindow w`（游戏主界面）。注意游戏主界面此时只是创建，并不显示。

关键的 `connect` 建立了一个信号槽连接。当 `s`（开始界面）发出 `startGameClicked` 信号时，槽函数——一个 Lambda 表达式——执行 `w.show()`，显示游戏主窗口。Lambda 中的 `[&]` 按引用捕获外部变量，这样能直接访问 `w`。

**`Q_OBJECT` 宏**是所有使用信号槽的类的必需品。你在类中写了 `signals` 和 `slots`，Qt 的元对象编译器（MOC）会识别这些关键字并在幕后生成额外的代码。如果忘记加 `Q_OBJECT`，编译会报一大堆看不懂的错误。

**`private slots:`** 块声明了两个槽函数。槽函数就是普通的成员函数，只不过它们可以被信号触发调用。`explicit` 关键字修饰构造函数，防止隐式类型转换。

```cpp
    connect(btnExit,  &QPushButton::clicked, this, &StartWindow::onExitClicked);
```

构造函数里完成了整个开始界面的搭建。

接着逐个创建界面元素——一个 `QLabel` 显示标题文字（40号加粗字体，居中对齐），两个 `QPushButton` 分别显示"开始游戏"和"退出游戏"（最小尺寸 200×60）。

所有这些控件都**堆在堆上分配**（`new`），并将 `this` 作为父对象传入。Qt 的父子对象机制会负责内存管理——当窗口 `StartWindow` 销毁时，所有子控件会自动被 `delete`，你不需要手动释放。

**布局系统**是让界面自适应窗口大小的关键。`QVBoxLayout` 是一个垂直布局器，它按从上到下的顺序排列控件：

- `addStretch(1)` 在顶部放一个弹性空间（权重 1），占据一定比例的空隙
- `addWidget(title)` 放置标题
- `addSpacing(50)` 插入 50 像素的固定间隙
- `addWidget(btnStart, 0, Qt::AlignCenter)` 放置开始按钮并居中
- `addSpacing(20)` 再插 20 像素
- `addWidget(btnExit, 0, Qt::AlignCenter)` 放置退出按钮并居中
- `addStretch(2)` 底部弹性空间（权重 2），比例是顶部的两倍

`setLayout(layout)` 将布局器安装到窗口上。从此，无论窗口怎么放大缩小，控件都会自动居中且间距均匀。

最后两行 `connect` 将按钮的点击信号连接到对应的槽函数。

```cpp
#define DOLL_H
```

`Doll` 继承自 `QObject` 而非 `QWidget`，因为它**不是一个可视控件**——它只存储游戏数据和逻辑，不负责把自己画出来。画人偶这件事由 `MainWindow` 的绘图函数代劳。

**成员变量一览：**

- `x, y`：人偶在 9×9 网格中的坐标。注意这里没有用 QPoint，而是直接两个 int，让外部访问更直接
- `player`：1 还是 2，决定颜色和起始位置
- `hp`：血量，初始 100，归零即死亡

- `isMoving`：一个状态标记，true 表示人偶正在棋盘上移动
- `arrowCount`：移动过程中累计碰到的箭头数，它直接决定了命中时造成的伤害

**`currentDir` 和 `launchDir` 为什么要分开？** 这是为了区分"玩家意图"和"实际运动"。玩家按 W 设为向上发射，但人偶在路上可能被箭头转向——`launchDir` 保留原始意图不被覆盖，而 `currentDir` 随动改变。

```cpp
void Doll::moveStep()

    }
}
```

## 七、gameboard.h —— 棋盘类头文件

```cpp
#ifndef GAMEBOARD_H
#define GAMEBOARD_H
#include <QObject>

enum Direction { Right, Down, Left, Up, DirCount };

struct Cell {
    bool hasArrow = false;
    Direction dir = Right;
    bool isBlock = false;
};

class GameBoard : public QObject
{
    Q_OBJECT
public:
    static const int SIZE = 9;
    explicit GameBoard(QObject *parent = nullptr);
    void initBoard();
    void rotateArrow(int x, int y);
    Cell getCell(int x, int y);

private:
    Cell board[SIZE][SIZE];
    const int ARROW_COUNT = 27;
};

#endif
```

**`enum Direction`** 定义了四个方向和一个特殊值 `DirCount`。`DirCount` 的值等于 4（枚举从 0 开始，第四个是 `Up`=3，`DirCount`=4），它除了作为方向总数，还用在旋转箭头的**取模运算**中。

**`struct Cell`** 用一个结构体表示棋盘上的一格。用了 C++11 的**成员默认初始化值**——`hasArrow = false` 等写法让新建的 Cell 自动处于"空白"状态。一个单元格可以同时是"无箭头且无障碍物"，也可以"有箭头"或"是障碍物"（但不能同时有箭头又是障碍物，这是 `initBoard` 的逻辑来保证的）。

**`static const int SIZE = 9`** 是**静态常量**，属于类本身而非实例，可以直接用 `GameBoard::SIZE` 访问。

**`board[SIZE][SIZE]`** 构成了 9×9 的棋盘核心。它是栈上分配的二维数组，81 个 Cell 直接嵌入 `GameBoard` 对象中，无需额外堆分配。

`ARROW_COUNT = 27` 控制每局游戏随机生成的箭头数量。

## 八、gameboard.cpp —— 棋盘类实现

```cpp
#include "gameboard.h"
#include <QRandomGenerator>

GameBoard::GameBoard(QObject *parent) : QObject(parent) {
    initBoard();
}

void GameBoard::initBoard() {
    for (int y = 0; y < SIZE; ++y)
        for (int x = 0; x < SIZE; ++x)
            board[y][x] = Cell();

    int cnt = 0;
    while (cnt < ARROW_COUNT) {
        int x = QRandomGenerator::global()->bounded(SIZE);
        int y = QRandomGenerator::global()->bounded(SIZE);
        if (!board[y][x].hasArrow && !(x==1&&y==1) && !(x==7&&y==7)) {
            board[y][x].hasArrow = true;
            board[y][x].dir = (Direction)QRandomGenerator::global()->bounded(DirCount);
            cnt++;
        }
    }

    for (int i = 0; i < 8; ++i) {
        int x = QRandomGenerator::global()->bounded(SIZE);
        int y = QRandomGenerator::global()->bounded(SIZE);
        if (!board[y][x].hasArrow)
            board[y][x].isBlock = true;
    }
}
```

`initBoard()` 分三步走：

**第一步**：双重循环把整个棋盘 81 格全部重置为 `Cell()`——干净的空白格。这是一种"暴力清零"策略，简单可靠。

**第二步**：用 `while` 循环随机生成 27 个箭头。`QRandomGenerator::global()->bounded(SIZE)` 产生 [0, 9) 的随机整数。关键条件 `!(x==1&&y==1) && !(x==7&&y==7)` 保证了两个玩家的出生点不会有箭头，避免"刚出生就被箭头带飞"。如果把枚举值强转成 `Direction`，得到随机的四个方向之一。

这里有潜在的**性能问题**：随着棋盘填满，`while` 循环找不到空位的时间会越来越长。但对于 81 格放 27 个箭头（占用率只有 1/3），碰撞概率很低，实际影响微小。

**第三步**：再随机放 8 个障碍物，条件是"该位置没有箭头"。这保证了同一格不会同时有箭头和障碍物。

```cpp
void GameBoard::rotateArrow(int x, int y) {
    if (x < 0 || x >= SIZE || y < 0 || y >= SIZE) return;
    if (board[y][x].hasArrow)
        board[y][x].dir = (Direction)((board[y][x].dir + 1) % DirCount);
}
```

**`rotateArrow`** 是最精妙的方法。`(board[y][x].dir + 1) % DirCount` 这一行完成了箭头的 90° 顺时针旋转：

- Right(0) → Down(1) → Left(2) → Up(3) → Right(0)

取模 `% DirCount` 让它在 3 后循环回 0，实现了一个完美闭合的方向环。`DirCount` 作为枚举的最后一个值，天然就是方向数量，这是一种精巧的设计。

```cpp
Cell GameBoard::getCell(int x, int y) {
    if (x < 0 || x >= SIZE || y < 0 || y >= SIZE) return Cell();
    return board[y][x];
}
```

`getCell` 返回的是一个 Cell 的**拷贝**而非引用。这意味着外部对返回值的修改不会影响棋盘——这是一种防御性编程，确保棋盘数据只能通过 `initBoard` 和 `rotateArrow` 来修改。

### 10.2 坐标转换系统

```cpp
QRect MainWindow::getBoardRect() const
{
    int w = width();
    int h = height() - m_boardMargin;
    int boardPixel = qMin(w, h);
    int x = (w - boardPixel) / 2;
    int y = m_boardMargin / 2;
    return QRect(x, y, boardPixel, boardPixel);
}

int MainWindow::getCellSize() const
{
    return getBoardRect().width() / 9;
}

QPoint MainWindow::gridToPos(int x, int y) const
{
    QRect br = getBoardRect();
    int cs = getCellSize();
    return QPoint(br.x() + x * cs + cs/2, br.y() + y * cs + cs/2);
}

QPoint MainWindow::posToGrid(const QPoint &pos) const
{
    QRect br = getBoardRect();
    int cs = getCellSize();
    if (cs <= 0) return QPoint(0,0);

    int gx = (pos.x() - br.x()) / cs;
    int gy = (pos.y() - br.y()) / cs;
    gx = qBound(0, gx, 8);
    gy = qBound(0, gy, 8);
    return QPoint(gx, gy);
}
```

这里的核心设计思想是**棋盘始终是正方形**且**居中显示**。`qMin(w, h)` 取窗口宽高中的较小值作为棋盘边长，然后居中放置。这让窗口不管怎么缩放，棋盘都不会变形。

`getCellSize()` 返回单个格子的像素大小，等于棋盘总宽度除以 9。因为棋盘是正方形，宽高被 9 整除时格子也是正方形的。

坐标转换中 `cs/2` 的偏移是为了得到格子的**中心**坐标——这样画出来的圆形人偶才会位于格子正中间，而不是贴着左上角。

`qBound(0, gx, 8)` 等价于 `qMax(0, qMin(gx, 8))`，确保结果在 0 到 8 之间。这是防御性编程，防止点击棋盘外部时坐标越界。

### 10.3 渲染系统

```cpp

```

`paintEvent` 是整个渲染的调度中心。`Q_UNUSED(e)` 告诉编译器不要因为参数未使用而报警告。

`p.setRenderHint(QPainter::Antialiasing)` 开启了**抗锯齿**，让人偶的圆形和棋盘线条更平滑。

绘制顺序很重要：棋盘在底层，人偶在中间，血条在顶层。这是经典的**画家算法**——先画的内容在下层，后画的内容覆盖在上面。

`QString(...).arg(curPlayer).arg(turnTime)` 使用了 Qt 的**字符串模板替换**，每个 `%1`、`%2` 依次被 `arg()` 的值替换。

```cpp
void MainWindow::drawBoard(QPainter &p)
{
    QRect br = getBoardRect();
    int cs = getCellSize();

    for (int y = 0; y < 9; ++y) {
        for (int x = 0; x < 9; ++x) {
            QRect r(br.x() + x*cs, br.y() + y*cs, cs, cs);
            p.setPen(Qt::black);
            p.setBrush((x + y) % 2 ? Qt::white : QColor(240,240,240));
            p.drawRect(r);

            Cell c = board->getCell(x, y);
            if (c.isBlock) {
                p.setBrush(Qt::darkGray);
                p.drawRect(r.adjusted(4,4,-4,-4));
            }
            if (c.hasArrow) {
                p.setPen(Qt::blue);
                QPoint ct = r.center();
                int len = cs * 0.25;
                switch (c.dir) {
                case Right: p.drawLine(ct, ct + QPoint(len,0)); break;
                case Down:  p.drawLine(ct, ct + QPoint(0,len)); break;
                case Left:  p.drawLine(ct, ct - QPoint(len,0)); break;
                case Up:    p.drawLine(ct, ct - QPoint(0,len)); break;
                }
            }
        }
    }
}
```

`(x + y) % 2` 用来产生棋盘格子的明暗交替效果——像国际象棋棋盘一样。

障碍物用 `r.adjusted(4,4,-4,-4)` 在格子内缩进 4 像素画一个深灰色方块，留出边界空隙让它看起来更自然。

箭头绘制用 `QPoint` 的算术运算：`ct + QPoint(len, 0)` 表示从格子中心向右画一条长为 `len` 的线。箭头长度是格子的 25%，视觉上清晰而不突兀。

```cpp
}
```

人偶用一个圆形 `drawEllipse` 绘制，半径是格子的 20%，刚好不算太大。颜色区分：红色是玩家 1，蓝色是玩家 2。

血条使用了**双层矩形**技术：底层是浅灰背景，上层是彩色前景，前景的宽度和血量成正比。`barW * doll1->hp / 100` 算出当前血量对应的像素宽度。血量满时全覆盖，血量减少时前景变短——这是一个简洁却直观的 HP 条实现。

玩家 2 的血条用 `w - barW - 20` 放在窗口右侧，与玩家 1 的血条形成左右对称。

### 10.4 输入处理

```cpp
void MainWindow::mousePressEvent(QMouseEvent *e)
{
    if (gameEnd || moveTimer->isActive()) return;
    QPoint g = posToGrid(e->pos());
    board->rotateArrow(g.x(), g.y());
    update();
}

void MainWindow::keyPressEvent(QKeyEvent *e)
{
    if (gameEnd) return;

    if (e->key() == Qt::Key_W && !moveTimer->isActive())
        currDoll->setLaunchDirection(Up);
    if (e->key() == Qt::Key_S && !moveTimer->isActive())
        currDoll->setLaunchDirection(Down);
    if (e->key() == Qt::Key_A && !moveTimer->isActive())
        currDoll->setLaunchDirection(Left);
    if (e->key() == Qt::Key_D && !moveTimer->isActive())
        currDoll->setLaunchDirection(Right);

    if (e->key() == Qt::Key_Space && !moveTimer->isActive())
        launch();

    update();
}
```

鼠标点击事件处理得很简洁：游戏结束或在移动中时直接忽略（`return`）。否则把鼠标坐标转成网格坐标，调用 `rotateArrow` 旋转该格的箭头，然后 `update()` 触发重绘。

键盘处理中反复出现的 `!moveTimer->isActive()` 条件是为了在移动动画播完之前锁定输入，防止玩家在人偶移动时反复改变方向。

### 10.5 移动与碰撞逻辑

```cpp
void MainWindow::launch()
{
    currDoll->arrowCount = 0;
    currDoll->applyLaunchDirection();
    currDoll->isMoving = true;
    moveTimer->start();
}

void MainWindow::updateMoveAnimation()
{
    int oldX = currDoll->x;
    int oldY = currDoll->y;
    currDoll->moveStep();
    int x = currDoll->x;
    int y = currDoll->y;

    if (x < 0 || x >= 9 || y < 0 || y >= 9) {
        currDoll->x = oldX;
        currDoll->y = oldY;
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }

    Cell cell = board->getCell(x, y);
    Doll* enemy = (currPlayer == 1) ? doll2 : doll1;

    if (cell.isBlock) {
        currDoll->x = oldX;
        currDoll->y = oldY;
        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }

    if (x == enemy->x && y == enemy->y) {
        currDoll->x = oldX;
        currDoll->y = oldY;

        int damage = currDoll->arrowCount * 2;
        enemy->hurt(damage);

        QMessageBox::information(this, "命中",
            QString("途经箭头:%1 → 造成伤害:%2")
                .arg(currDoll->arrowCount).arg(damage));

        if (doll1->isDead()) gameOver(2);
        if (doll2->isDead()) gameOver(1);

        currDoll->isMoving = false;
        moveTimer->stop();
        switchPlayer();
        return;
    }

    if (cell.hasArrow) {
        currDoll->currentDir = cell.dir;
        currDoll->arrowCount++;
    }

    update();
}
```

这是整个游戏逻辑的**心脏**——每 180 毫秒执行一次，处理人偶移动的全流程。

**保存旧坐标**（oldX, oldY）是关键设计模式——任何时候导致移动终止（出界、撞墙、命中），都把坐标回退到移动前的位置。这意味着人偶"在那一格停了下来"，而不是停在障碍物或边界上。

**伤害公式** `arrowCount * 2` 的设计很有游戏性：途经的箭头越多，伤害越高。这鼓励玩家在发射前先规划路线，尽量让路径穿过更多箭头。

**命中后的三段式处理**：扣血 → 判断胜负 → 切换玩家，每一步都有明确职责。

**箭头处理**放在最后：如果格子上有箭头，改变移动方向并累计箭头计数。这不终止移动——人偶继续沿新方向前进。

### 10.6 游戏流程控制

```cpp
void MainWindow::updateTurnTimer()
{
    if (gameEnd || moveTimer->isActive()) return;
    if (--turnTime <= 0)
        switchPlayer();
    update();
}

void MainWindow::switchPlayer()
{
    currPlayer = (currPlayer == 1) ? 2 : 1;
    turnTime = 60;
    currDoll = (currPlayer == 1) ? doll1 : doll2;
    update();
}

void MainWindow::gameOver(int winner)
{
    gameEnd = true;
    QMessageBox::information(this, "游戏结束", QString("玩家%1 胜利！").arg(winner));
    QMessageBox::information(this, "重置", "确定后重新开始");

    doll1->reset();
    doll2->reset();
    board->initBoard();
    gameEnd = false;
    currPlayer = 1;
    turnTime = 60;
    update();
}
```

`--turnTime <= 0` 使用了**前置递减**，先减后判断。当时间归零时自动切换玩家。

`switchPlayer` 是状态切换的枢纽：交换玩家编号，重置时间为 60 秒，更新当前人偶指针。

`gameOver` 的处理很有仪式感——弹出两次对话框，让玩家清楚地看到胜负结果。然后完全重置所有状态：两个 Doll 的 `reset()`，棋盘的 `initBoard()`，状态变量全部回初始值。这保证了下一局从一个完全干净的状态开始。
